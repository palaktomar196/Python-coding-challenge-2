Question:- second largest element 






Answer:-  def second_largest(lst):
    largest = second = -999999
    
    for i in lst:
        if i > largest:
            second = largest
            largest = i
        elif i > second and i != largest:
            second = i
    return second


nums = list(map(int, input("Enter numbers: ").split(",")))
print(second_largest(nums))
